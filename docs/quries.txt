1Ô∏è‚É£5Ô∏è‚É£ WITH VARIABLES (Best Practice)
Create Student with Variables
curl -X POST http://localhost:4000/graphql \
-H "Content-Type: application/json" \
-d '{
  "query": "mutation CreateStudent($name: String!, $email: String!, $rollNumber: String!, $age: Int, $phone: String, $departmentId: ID!) { addStudent(name: $name, email: $email, rollNumber: $rollNumber, age: $age, phone: $phone, departmentId: $departmentId) { id name email rollNumber department { name } } }",
  "variables": {
    "name": "Meena Kumari",
    "email": "meena@example.com",
    "rollNumber": "BCA006",
    "age": 20,
    "phone": "9876543215",
    "departmentId": "YOUR_DEPARTMENT_ID"
  }
}'
Get Student with Variables
curl -X POST http://localhost:4000/graphql \
-H "Content-Type: application/json" \
-d '{
  "query": "query GetStudent($studentId: ID!) { student(id: $studentId) { id name email rollNumber department { name code hod } } }",
  "variables": {
    "studentId": "YOUR_STUDENT_ID"
  }
}'
Update Student with Variables
curl -X POST http://localhost:4000/graphql \
-H "Content-Type: application/json" \
-d '{
  "query": "mutation UpdateStudent($id: ID!, $name: String, $age: Int, $phone: String) { updateStudent(id: $id, name: $name, age: $age, phone: $phone) { id name email age phone } }",
  "variables": {
    "id": "YOUR_STUDENT_ID",
    "name": "Updated Name",
    "age": 22,
    "phone": "8888888888"
  }
}'
